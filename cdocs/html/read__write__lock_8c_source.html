161 ReadWriteLock_t *CreateReadWriteLockPreferReader(void)
162 {
163  RWLockPreferReader_t *lock;
164 
165  lock = (RWLockPreferReader_t *)malloc(sizeof(RWLockPreferReader_t));
166  if (!lock) {
167  /*
168  * If the initial malloc fails, just get out.
169  */
170  return NULL;
171  }
172 
173  lock->Type = RwPreferReader;
174 
175  lock->ReadCount = 0;
176 
177  /*
178  * Preload these with NULLs to allow simplified error handling.
179  */
180  lock->ReadLock = NULL;
181  lock->ResourceLock = NULL;
182 
183 
184  lock->ReadLock = xSemaphoreCreateMutex();
185  if (lock->ReadLock == NULL) {
186  goto ERROR;
187  }
188 
189  /*
190  * ResourceLock CANNOT be a mutex. In FreeRTOS, as in most OS's,
191  * a thread is not allowed to unlock another thread's mutex. But
192  * the very nature of a Reader Lock allows an arbitrary ordering
193  * of unlocks when multiple threads hold the reader lock.
194  * Semaphores are not subject to this constraint.
195  */
196  lock->ResourceLock = xSemaphoreCreateBinary();
197  if (lock->ResourceLock == NULL) {
198  goto ERROR;
199  }
200 
201  /*
202  * Initialize it as "full", so it behaves similar to a mutex.
203  */
204  xSemaphoreGive(lock->ResourceLock);
205 
206  return (ReadWriteLock_t *)lock;
207 
208 
209  /*
210  * Handle most error cases here. This simplifies cleanup
211  * and is a common pattern in most OSs.
212  */
213 ERROR:
214  if (!lock->ReadLock) {
215  vSemaphoreDelete(lock->ReadLock);
216  }
217 
218  if (!lock->ResourceLock) {
219  vSemaphoreDelete(lock->ResourceLock);
220  }
221 
222  free(lock);
223  return NULL;
224 }

325 void FreeReadWriteLock(ReadWriteLock_t *lock)
326 {
327  RWLockPreferReader_t *lockR = (RWLockPreferReader_t *)lock;
328  RWLockPreferWriter_t *lockW = (RWLockPreferWriter_t *)lock;
329 
330  configASSERT(lockR != NULL);
331 
332  if (lockR->Type == RwPreferReader) {
333  vSemaphoreDelete(lockR->ReadLock);
334  vSemaphoreDelete(lockR->ResourceLock);
335  free(lockR);
336  }
337  else if (lockW->Type == RwPreferWriter) {
338  vSemaphoreDelete(lockW->ReadLock);
339  vSemaphoreDelete(lockW->WriteLock);
340  vSemaphoreDelete(lockW->ResourceLock);
341  vSemaphoreDelete(lockW->BlockReadersLock);
342  free(lockW);
343  }
344  else {
345  configASSERT(!"Bad RW Lock type!");
346  }
347 }

350 void ReaderLock(ReadWriteLock_t *lock)
351 {
352  RWLockPreferReader_t *lockR = (RWLockPreferReader_t *)lock;
353  RWLockPreferWriter_t *lockW = (RWLockPreferWriter_t *)lock;
354 
355  configASSERT(lockR != NULL);
356 
357  if (lockR->Type == RwPreferReader) {
358 
359  xSemaphoreTake(lockR->ReadLock, portMAX_DELAY);
360 
361  lockR->ReadCount++;
362  if (lockR->ReadCount == 1) {
363  xSemaphoreTake(lockR->ResourceLock, portMAX_DELAY);
364  }
365 
366  xSemaphoreGive(lockR->ReadLock);
367  }
368  else if (lockW->Type == RwPreferWriter) {

370  xSemaphoreTake(lockW->BlockReadersLock, portMAX_DELAY);
371  xSemaphoreTake(lockW->ReadLock, portMAX_DELAY);
