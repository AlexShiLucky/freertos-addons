78 void MemoryPool::CalculateValidAlignment()
79 {
83  if (Alignment < (int)sizeof(unsigned char *)) {
84  Alignment = (int)sizeof(unsigned char *);
85  }
86 
87  int alignmentBit = 0x1;
88  int i;
89 
90  for (i = 0; i < 31; i++) {
91  if (Alignment == alignmentBit) {
92  break;
93  }
94  alignmentBit <<= 1;
95  }
96 
97  if (i >= 31) {
98 #ifndef CPP_FREERTOS_NO_EXCEPTIONS
99  throw MemoryPoolBadAlignmentException();
100 #else
101  configASSERT(!"MemoryPool Bad Alignment");
102 #endif

107 void MemoryPool::CalculateItemSize()
108 {
109  if (ItemSize <= Alignment) {
110 
111  ItemSize = Alignment;
112  }
113  else {
114 
115  int alignmentCount = ItemSize / Alignment;

118  }

120  ItemSize = alignmentCount * Alignment;
